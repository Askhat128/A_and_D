function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);

  var row = [new Date()];
  for (var key in data) {
    row.push(data[key]);
  }

  sheet.appendRow(row);
  return ContentService.createTextOutput("Спасибо! Данные успешно отправлены.").setMimeType(ContentService.MimeType.TEXT);
}
